```bash
#COMANDOS GIT BASH
git init # iniciar un historial de cambios
git add . # Agregar archivos para que genere el historial de cambios
git commit -am "comentario" # agregar un comentario de los cambios que has hecho
git push # subir los cambios al repo (github/gitlab/otro)

git remote # agregar donde los subiras (que repo usaras y que protocolo https o ssh)
git log # ver cambios
git log --stat # ver estadisticas
git log --graph # ver grafica de cambios
git config # configuracion
```
